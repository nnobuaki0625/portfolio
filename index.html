Scraping

スクレイピング

某経済情報サイトへスクレイピングするプログラムです。

テキスト情報の取得から、調整したフォーマットでExcelデーターとして書き出しを行います。

実行環境はGoogle Colabです。

詳しくはTAP

Scraping Code

Pythonで動くコードを載せています。

0から書いて実際に使用していますが、環境によっては機能しないと思います🙏

import requests from bs4 import BeautifulSoup import re import openpyxl star_regex = re.compile(r'★+') load_url = " ここにスクレイピングしたいURLを入れます " html = requests.get(load_url) soup = BeautifulSoup(html.content, "html.parser") data = [] # Create an empty list future = soup.find(class_="tbody") for element in soup.find_all(class_="future"): text_content = element.text # 改行でテキストを分割 lines = text_content.split('\n') # 各行を処理 for line in lines: # 正規表現を使用して★までの文字列を抽出 matches = re.search(r'.*★', line) if matches: extracted_text = matches.group() # 処理したテキストデーターをもう一度リスト化 moji = extracted_text.split('\n') for line in moji: parts = re.split('(日本|中国|米国|イギリス|メキシコ|ユーロ|オーストラリア|トルコ|香港|シンガポール|スイス|ドイツ|南アフリカ|フランス|カナダ)', line) if len(parts) > 2: mo = star_regex.search(parts[2]) if mo: parts.append(mo.group()) second_parts = parts[2] matches = re.sub('★+', ' ', second_parts) parts[2] = matches parts[1], parts[2], parts[3] = parts[3], parts[1], parts[2] data.append(parts) # Append the processed data to the list # Create Excel file and write data to it workbook = openpyxl.Workbook() sheet = workbook.active for row in data: sheet.append(row) workbook.save('output.xlsx') print("出力が完了したよ！")
